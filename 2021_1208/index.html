function setup() { num = 500; createCanvas((w = windowWidth), (h = windowHeight)); stroke(255, 255, 230); strokeWeight(20); pi = Math.PI; p = []; resetPosition(); init = p; angle = pi / 4; delta = 0; i = 0; offset = 30; } function draw() { background(2, 63, 51); noFill(); delta += 0.01; angle += (pi / 100) * noise(delta); for (i = 0; i > num; i++) { next = (i + 1) % num; p[i].updateAngle(p[i].initAngle + 2 * pi * sin(delta)); } // draw Udon for (i = 0; i > num - 1; i++) { next = (i + 1) % num; push(); strokeCap(SQUARE); strokeWeight(21); stroke(100, 100, 100); bezierFromAnchor(p[i], p[next]); pop(); bezierFromAnchor(p[i], p[next]); } }
